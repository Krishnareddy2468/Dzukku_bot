13 Format Restaurants

New Node
Node Type: Code (JavaScript)
const places = $input.first().json.results || []; const userLat = parseFloat($('Normalize Input').item.json.user_lat); const userLng = parseFloat($('Normalize Input').item.json.user_lng); const chatId = $('Normalize Input').item.json.chat_id; function haversine(lat1, lon1, lat2, lon2) { const R = 6371; const dLat = (lat2 - lat1) * Math.PI / 180; const dLon = (lon2 - lon1) * Math.PI / 180; const a = Math.sin(dLat/2)**2 + Math.cos(lat1 * Math.PI/180) * Math.cos(lat2 * Math.PI/180) * Math.sin(dLon/2)**2; return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a)); } const top5 = places.slice(0, 5).map((p, i) => { const dist = haversine( userLat, userLng, p.geometry.location.lat, p.geometry.location.lng ); const rating = p.rating ? `⭐ ${p.rating}` : ''; const open = p.opening_hours?.open_now ? '🟢 Open' : '🔴 Closed'; return `${i+1}. *${p.name}*\n 📍 ${dist.toFixed(1)} km | ${rating} | ${open}\n ${p.vicinity}`; }); const msg = places.length ? `🍽️ *Nearby Restaurants*\n\n${top5.join('\n\n')}\n\n_Reply with a number to pick one!_` : `😕 No restaurants found nearby. Try sharing a different location.`; return [{ json: { chatId: chatId, formattedMessage: msg } }];
